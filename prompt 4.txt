I built a Student Toolkit web application using HTML, CSS, and JavaScript. 
One feature is an Idea Board where users can add ideas as sticky notes.

Problem:
Currently the system inserts user input using innerHTML when creating the sticky note. 
Because of this, if a user enters HTML or JavaScript code such as:

<b>Important</b>
<script>alert("Hello")</script>

the browser interprets it as actual HTML or script instead of plain text.

What I want:
The system should treat all user input as plain text and NOT execute or render HTML tags.

Requirements:
1. If a user enters HTML tags like <b>Important</b>, it should display exactly as text, not bold formatting.
2. If someone enters script tags like <script>alert("Hello")</script>, the script must never run.
3. Prevent any XSS security issues.
4. Keep the existing UI design exactly the same.
5. Only modify the JavaScript logic inside the addIdea() function.

Suggested approach:
Instead of using innerHTML to insert user input, use safer methods such as textContent or createTextNode.

I will provide my full HTML, CSS, and JavaScript code below. 
Please update only the necessary parts and return the corrected code.

<script>

function showSection(id){
document.querySelectorAll('.page-section').forEach(s=>s.classList.remove('active'));
document.getElementById(id).classList.add('active');
}

window.onload=()=>showSection('ideas');

const colors=['note-color-1','note-color-2','note-color-3','note-color-4'];

function updateCounter(){

const input=document.getElementById("ideaInput");
const counter=document.getElementById("charCounter");

const length=input.value.length;

counter.textContent=length+" / 200";

if(length>=200){
counter.classList.add("limit");
}else{
counter.classList.remove("limit");
}

input.classList.remove("input-error");
}

function addIdea(){

const input=document.getElementById("ideaInput");
const grid=document.getElementById("ideaGrid");

const text=input.value.trim().toLowerCase();

if(text===""){
input.classList.add("input-error");
return;
}

const existing=document.querySelectorAll("#ideaGrid .sticky-note p");

for(let idea of existing){
if(idea.textContent.trim().toLowerCase()===text){
input.classList.add("input-error");
return;
}
}

const div=document.createElement("div");

const randomColor=colors[Math.floor(Math.random()*colors.length)];

div.className="sticky-note "+randomColor;

div.innerHTML=
`
<p>${input.value}</p>
<button class="delete-note" onclick="this.parentElement.remove()">
<i class="ph ph-trash"></i>
</button>
`;

grid.prepend(div);

input.value="";
updateCounter();

}

let totalTime=1500;
let timeLeft=totalTime;
let timer=null;

function updateDisplay(){

let m=Math.floor(timeLeft/60);
let s=timeLeft%60;

document.getElementById("timerDisplay").textContent=
String(m).padStart(2,'0')+":"+String(s).padStart(2,'0');

}

function startTimer(){
if(timer)return;

timer=setInterval(()=>{

if(timeLeft>0){
timeLeft--;
updateDisplay();
}else{
clearInterval(timer);
timer=null;
alert("Session Completed!");
}

},1000);
}

function pauseTimer(){
clearInterval(timer);
timer=null;
}

function resetTimer(){
pauseTimer();
timeLeft=totalTime;
updateDisplay();
}

function calculateGPA(){

let pts=0;
let crd=0;

document.querySelectorAll("#courseTable tr").forEach(row=>{

let c=parseFloat(row.querySelector(".credit")?.value)||0;
let g=parseFloat(row.querySelector(".grade")?.value)||0;

pts+=c*g;
crd+=c;

});

let gpa=crd===0?0:(pts/crd).toFixed(2);

let res=document.getElementById("gpaResult");

res.style.display="block";
res.textContent="GPA: "+gpa;

}
